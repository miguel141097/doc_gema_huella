def incrementarPreciosWithPorcentaje

    valor = @menuDietetico.max.huellaNutricional
    incremento = 1
    
    # En funcion del valor de huella nutricional se hace un incremento en porcentaje
    if (valor < 1)
        incremento = 1.15
    elsif (valor < 2)
        incremento = 1.30
    else
        incremento = 1.45
    end
    
    nuevoPrecio = @precioDietetico.map{|precio| (precio * incremento).round(2)}

end

def mediaHuellaNutricional
    huella = @menuDietetico.map{|plato| plato.huellaNutricional}

    dividendo = huella.reduce(:+)
    divisor = (huella.size) * 1.0

    (dividendo / divisor).round(2)
end
